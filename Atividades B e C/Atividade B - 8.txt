CREATE PROCEDURE TESTE_SINQIA (
    @FUNCAO VARCHAR(10),
    @COD_CIDADE INT,
    @CID_NOME VARCHAR(100),
    @CID_ESTADO CHAR(2)
)
AS
BEGIN
    IF @FUNCAO = 'INSERT'
    BEGIN
        INSERT INTO CIDADES (COD_CIDADE, CID_NOME, CID_ESTADO)
        VALUES (@COD_CIDADE, @CID_NOME, @CID_ESTADO);
    END
    ELSE IF @FUNCAO = 'UPDATE'
    BEGIN
        UPDATE CIDADES
        SET CID_NOME = @CID_NOME,
            CID_ESTADO = @CID_ESTADO
        WHERE COD_CIDADE = @COD_CIDADE;
    END
    ELSE IF @FUNCAO = 'DELETE'
    BEGIN
        DELETE FROM CIDADES
        WHERE COD_CIDADE = @COD_CIDADE;
    END
END;
